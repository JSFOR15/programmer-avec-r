\begin{reponse}{3.6}

\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> max(x)
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> mean(x[1:5])
> mean(head(x, 5))
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> mean(x[16:20])
> mean(x[(length(x) - 4):length(x)])

\end{Schunk}
    \end{enumerate}
  
\end{reponse}
