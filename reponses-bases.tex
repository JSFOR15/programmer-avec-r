\section*{Chapitre \ref{bases}}
\begin{reponse}{2.1}
    Soit \code{x} le nom de la liste.
    \begin{enumerate}
\item
\begin{Schunk}
\begin{Sinput}
> list(1:5, data = matrix(1:6, 2, 3), numeric(3),
+     test = logical(4))
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> names(x)
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> mode(x$test)
> length(x$test)
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> dim(x$data)
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> x[[2]][c(2, 3)]
\end{Sinput}
\end{Schunk}
\item
\begin{Schunk}
\begin{Sinput}
> x[[3]] <- 3:8
\end{Sinput}
\end{Schunk}
    \end{enumerate}
  
\end{reponse}
